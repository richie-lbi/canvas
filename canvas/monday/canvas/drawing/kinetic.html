KineticJS > Drawing

  1. Create a stage - takes a pseudo canvas elements
  2. Draw a shape
  3. Add to Layer (- )each layer is its own canvas)
  4. Don't forgt to update stage
  5. Style Object notation
Drawing the 'L' 
var lbi = new Kinetic.Layer();
var l = new Kinetic.Shape({
drawFunc: function(canvas) {
  var context = canvas.getContext();
  context.beginPath();
  context.moveTo(0, 0);
  context.lineTo(11,0);
  context.lineTo(11,35);
  context.lineTo(30,35);
  context.lineTo(30,44);
  context.lineTo(0,44);
  context.closePath();
  canvas.fillStroke(this);
},
fill: '#f1001c'
});

lbi.add(l);

stage.add(lbi);
